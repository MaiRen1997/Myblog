## 全局二次封装方法(最优)

全局修改element组件的方法，将触发方法套上一层防抖

在main的js中使用如下代码

```js
import { throttle } from 'lodash-es'
// 防止按钮连续点击, 0.5s内只能点一次, 点击后自动失去焦点
ElementUI.Button.methods.handleClick = throttle(
  function(evt) {
    let target = evt.target
    if (target.nodeName == 'SPAN') {
      target = evt.target.parentNode
    }
    target.blur()
    this.$emit('click', evt);
  },
  500,
  {
    trailing: false
  }
)
```

在e-button中使用

```html
<el-button v-no-more-click type="primary" >保存</el-button>
```

## 单个使用自定义指令

```js
//在main.js中
// 提交以后禁用按钮一段时间，防止重复提交
import Vue from 'vue'
Vue.directive('noMoreClick', {
  inserted(el, binding) {
    el.addEventListener('click', e => {
      el.classList.add('is-disabled')
      el.disabled = true
      setTimeout(() => {
        el.disabled = false
        el.classList.remove('is-disabled')
      }, 2000)//我这里设置的是2000毫秒也就是2秒
    })
  }
})
```



## 二次封装el-button

对el-button二次封装保留原本参数

### 创建全局防抖指令

```js
Vue.directive('delay', {
  inserted: function(el) {
    el.addEventListener('click', () => {
      el.disabled = true
      setTimeout(() =>{
        el.disabled = false
      },2000);
    })
  }
});
```

### 创建组件delayButton.vue使用render渲染按钮

```js
<script>
  /**
   * @Description: 防抖按钮一键生成
   * @author Mr.Long
   * @date 2021/1/14
  */
  export default {
    name: 'MenuItem',
    functional: true,
    /**
     * 属性跟element按钮一样
     * element-ui: https://element.eleme.cn/#/zh-CN/component/button
    */
    props: {
      //朴素按钮
      plain:{
        type: Boolean,
        default: false
      },
      //是否禁用
      loading:{
        type: Boolean,
        default: false
      },
      //尺寸
      size: {
        type: String,
        default: 'small'
      },
      //是否禁用
      round:{
        type: Boolean,
        default: false
      },
      //是否禁用
      disabled:{
        type: Boolean,
        default: false
      },
      //按钮类型
      icon: {
        type: String,
        default: ''
      },
      //按钮类型
      type: {
        type: String,
        default: 'primary'
      },
      //按钮标题
      title: {
        type: String,
        default: ''
      },
      //按钮点击事件
      buttonOnclick: {
        type: Function,
        default: () => {}
      },
    },
    //渲染按钮
    render(h, context) {
      const { icon, type, title,disabled,size,plain,loading, round,buttonOnclick } = context.props;
      return [<el-button type={type} plain={plain} loading={loading} size={size} round={round} disabled={disabled} icon={icon} v-delay on-click={buttonOnclick}>{title}</el-button>];
    }
  }
</script>
```

:::warning 注意

由于render需要重新绑定click事件所以我们需要把click方法像参数一样传进来

:::

### 在main.js中全局注册delay-button方便使用

```js
import delayButton from './components/delayButton'
Vue.component('delayButton', delayButton);
```

### 也可以覆盖原el-button

```js
// main.js

import DDButton from './components/delayButton';

// 覆盖 element button 解决点击按钮后，长按回车无限触发点击事件问题
Vue.component('el-button', DDButton);
```
