```js
//密码强度等级
getPwdLevel:function (pwd,minLength=8) {
    var level = 0;
    if (pwd.length < minLength) return level;
    if (/\d/.test(pwd)) level++; //数字
    if (/[a-z]/.test(pwd)) level++; //小写
    if (/[A-Z]/.test(pwd)) level++; //大写
    if (/\W/.test(pwd)) level++; //特殊字符
    return level;
}
```
