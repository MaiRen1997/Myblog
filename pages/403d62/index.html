# 数据类型转换

# int

使用int()函数：可以将浮点数或字符串转换为整数

float_number = 3.14
int_number = int(float_number)  # 结果为3
 
str_number = "42"
int_number = int(str_number)  # 结果为42

# float

使用float()函数：可以将整数或字符串转换为浮点数

int_number = 42
float_number = float(int_number)  # 结果为42.0
 
str_number = "3.14"
float_number = float(str_number)  # 结果为3.14
